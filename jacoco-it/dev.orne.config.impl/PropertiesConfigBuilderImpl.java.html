import java.io.File;
import java.net.URL;
import java.nio.file.Path;
import java.util.Map;
import java.util.Properties;

import org.apiguardian.api.API;

import dev.orne.config.PropertiesConfigBuilder;

/**
 * Implementation of {@code Properties} based immutable configuration builder.
 * 
 * @author <a href="https://github.com/ihernaez">(w) Iker Hernaez</a>
 * @version 1.0, 2025-05
 * @since 1.0
 * @see PropertiesConfigBuilder
 */
@API(status = API.Status.INTERNAL, since = "1.0")
public class PropertiesConfigBuilderImpl
extends AbstractConfigBuilderImpl<PropertiesConfigBuilder>
implements PropertiesConfigBuilder {

    /** The properties based configuration options. */
    protected final PropertiesConfigOptions propertyOptions;

    /**
     * Empty constructor.
     */
    public PropertiesConfigBuilderImpl() {
        super();
        this.propertyOptions = new PropertiesConfigOptions();
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public PropertiesConfigBuilder add(
            final Properties values) {
        this.propertyOptions.add(values);
        return thisBuilder();
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public PropertiesConfigBuilder add(
            final Map<String, String> values) {
        this.propertyOptions.add(values);
        return thisBuilder();
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public PropertiesConfigBuilder load(
            final String path) {
        this.propertyOptions.load(path);
        return thisBuilder();
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public PropertiesConfigBuilder load(
            final Path path) {
        this.propertyOptions.load(path);
        return thisBuilder();
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public PropertiesConfigBuilder load(
            final File file) {
        this.propertyOptions.load(file);
        return thisBuilder();
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public PropertiesConfigBuilder load(
            final URL url) {
        this.propertyOptions.load(url);
        return thisBuilder();
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public PropertiesMutableConfigBuilderImpl mutable() {
        return new PropertiesMutableConfigBuilderImpl(
                this.options,
                new MutableConfigOptions(),
                propertyOptions);
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public PropertiesConfigImpl build() {
        return new PropertiesConfigImpl(this.options, this.propertyOptions);
    }
}
