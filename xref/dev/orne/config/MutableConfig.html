25  import javax.validation.constraints.NotBlank;
26  import javax.validation.constraints.NotNull;
27  
28  import org.apiguardian.api.API;
29  
30  import dev.orne.config.impl.ConfigSubset;
31  
32  /**
33   * Configuration properties provider with properties mutable at runtime.
34   * 
35   * @author <a href="https://github.com/ihernaez">(w) Iker Hernaez</a>
36   * @version 1.0, 2019-07
37   * @version 2.0, 2025-04
38   * @since 0.1
39   * @see Config
40   */
41  @API(status = API.Status.STABLE, since = "1.0")
42  public interface MutableConfig
43  extends Config {
44  
45      /**
46       * Sets the value of the specified configuration property.
47       * 
48       * @param key The configuration property.
49       * @param value The value to set
50       * @throws ConfigException If an error occurs setting the configuration
51       * property value
52       */
53      void set(
54              @NotBlank String key,
55              String value);
56  
57      /**
58       * Sets the value of the configuration parameter.
59       * 
60       * @param key The configuration property.
61       * @param value The value to set
62       * @throws ConfigException If an error occurs setting the configuration
63       * property value
64       */
65      default void set(
66              @NotBlank String key,
67              Boolean value) {
68          set(key, value == null ? null : String.valueOf(value));
69      }
70  
71      /**
72       * Sets the value of the configuration parameter.
73       * 
74       * @param key The configuration property.
75       * @param value The value to set
76       * @throws ConfigException If an error occurs setting the configuration
77       * property value
78       */
79      default void set(
80              @NotBlank String key,
81              Integer value) {
82          set(key, value == null ? null : String.valueOf(value));
83      }
84  
85      /**
86       * Sets the value of the configuration parameter.
87       * 
88       * @param key The configuration property.
89       * @param value The value to set
90       * @throws ConfigException If an error occurs setting the configuration
91       * property value
92       */
93      default void set(
94              @NotBlank String key,
95              Long value) {
96          set(key, value == null ? null : String.valueOf(value));
97      }
98  
99      /**
100      * Removes the specified configuration properties.
101      * 
102      * @param keys The configuration properties.
103      * @throws ConfigException If an error occurs removing the configuration
104      * properties.
105      */
106     void remove(
107             @NotBlank String... keys);
108 
109     /**
110      * Creates a subset configuration containing only the properties
111      * with the specified prefix.
112      * 
113      * @param prefix The prefix for configuration keys.
114      * @return The subset configuration.
115      */
116     @Override
117     default @NotNull MutableConfig subset(
118             final @NotNull String prefix) {
119         return ConfigSubset.create(this, prefix);
120     }
121 }
